Deployment

Provides a Kubernetes Deployment.

Examples​

Create a deployment for nginx​

const labelApp = "myApp";

const deployment = provider.makeDeployment({
  properties: ({}) => ({
    metadata: {
      name: "myDeployment",
      namespace: "myNamespace"
      labels: {
        app: labelApp,
      },
    },
    spec: {
      replicas: 3,
      selector: {
        matchLabels: {
          app: labelApp,
        },
      },
      template: {
        metadata: {
          labels: {
            app: labelApp,
          },
        },
        spec: {
          containers: [
            {
              name: "nginx",
              image: "nginx:1.14.2",
              ports: [
                {
                  containerPort: 80,
                },
              ],
            },
          ],
        },
      },
    },
  }),
});
